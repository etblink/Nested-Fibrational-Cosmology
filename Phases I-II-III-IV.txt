        x'' + delta x' - x + x^3 = gamma*(cos(w1 t) + cos(w2 t))

    x = np.zeros(steps)

        return gamma*(np.cos(w1*tt) + np.cos(w2*tt))

    return t, x, v

def simulate_selector_duffing_fast(w1=1.0, w2=1.618, gamma=0.28, delta=0.22, 
                                  alpha=3.0, eps=0.03, T=520.0, dt=0.03, 
                                  x0=0.15, v0=0.0, y0=0.0, tau=40.0):
    steps = int(T/dt)
    t = np.linspace(0, T, steps)
    x = np.zeros(steps); v = np.zeros(steps); y = np.zeros(steps)
    x[0], v[0], y[0] = x0, v0, y0
    
    def u(tt):
        return gamma*(np.cos(w1*tt) + np.cos(w2*tt))
    
    u2_avg = 0.0
    lam = dt/tau
    
    for i in range(steps-1):
        tt = t[i]
        uu = u(tt)
        u2_avg = (1-lam)*u2_avg + lam*(uu*uu)
        y[i+1] = y[i] + dt*eps*(u2_avg - y[i])
        yi = y[i]
        v[i+1] = v[i] + dt*(-delta*v[i] + (1+alpha*yi)*x[i] - x[i]**3 + uu)
        x[i+1] = x[i] + dt*v[i+1]
    return t, x, y
